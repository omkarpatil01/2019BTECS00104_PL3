Code in any c++ language that can find an armstrong number from a given range.

When the sum of the cube of the individual digits of a number is equal to that number, the number is called Armstrong number. For example 153. Sum of its divisor is 13 + 53;+ 33; = 1+125+27 = 153

Sample code

            
  #include 
  #include 
  using namespace std;

  int main()
  {
      int num, r, sum, t, mm;
      int sno, eno;
      cout << "\n\n Find the Armstrong number for a given range of number:\n";
      cout << " Input starting number of range: ";
      cin >> sno;
      cout << " Input ending number of range: ";
      cin >> eno;
      cout << " Armstrong numbers in given range are: " << endl;
      for (num = sno; num <= eno; num++) 
      {
          t = num;
          sum = 0;
          while (t != 0) 
          {
              r = t % 10;
              mm = pow(r, 3);
              sum = sum + mm;
              t = t / 10;
          }
          if (sum == num)
              cout << num << " ";
      }
      cout << endl;
  }
